clc
clear all
close all

Ts = 0.01;
t = 0:Ts:1;
x_t = 18*sin(2*pi*7*t);
N = length(t); n = 0:N-1;
x_s = 18*sin(2*pi*7*n*Ts);

figure
plot(t,x_t)
xlabel('Time (seconds)')
ylabel('x(t)')
title('Apmlitude versus time')

figure
plot(n,x_s)
xlabel('n')
ylabel('x(n)')
title('Amplitude versus n')

y = fft(x_t);
fs = 1/Ts;
f = (0:length(y)-1)*fs/length(y);
figure;
plot(f,abs(y))
xlabel('Frequency (Hz)')
ylabel('Magnitude')
title('Magnitude (using FFT command)')
